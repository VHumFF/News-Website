import React, { useEffect, useState } from "react";
import BreakingNews from "@/components/BreakingNews";
import NewsList from "@/components/NewsList";
import LatestNews from "@/components/LatestNews";
import { Box, CircularProgress, Typography } from "@mui/material";

const DEFAULT_NEWS = {
  breakingNews: { image: "#", title: "Breaking News", description: "Text text text text..." },
  articles: [],
  latestNews: [],
};

export default function HomePage() {
  const [newsData, setNewsData] = useState(DEFAULT_NEWS);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const fetchData = async () => {
      setLoading(true);
      setError(null);
      try {
        const response = await fetch("/api/news");
        if (response.ok) {
          const data = await response.json();
          setNewsData({
            breakingNews: data.breakingNews || DEFAULT_NEWS.breakingNews,
            articles: data.articles || DEFAULT_NEWS.articles,
            latestNews: data.latest || DEFAULT_NEWS.latestNews,
          });
        } else {
          // setError("Failed to fetch news data.");
          setNewsData(DEFAULT_NEWS);
        }
      } catch (err) {
        // setError("An error occurred while fetching news data.");
        setNewsData(DEFAULT_NEWS);
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, []);

  if (loading) {
    return (
      <Box sx={{ display: "flex", justifyContent: "center", alignItems: "center", minHeight: "100vh" }}>
        <CircularProgress />
      </Box>
    );
  }

  if (error) {
    return (
      <Box sx={{ textAlign: "center", padding: 4 }}>
        <Typography variant="h6" color="error">
          {error}
        </Typography>
      </Box>
    );
  }

  return (
    <Box>
      {/* Breaking News */}
      <BreakingNews
        image={newsData.breakingNews.image}
        title={newsData.breakingNews.title}
        description={newsData.breakingNews.description}
      />

      {/* News List */}
      <NewsList news={newsData.articles} />

      {/* Latest News */}
      <LatestNews latestNews={newsData.latestNews} />
    </Box>
  );
}
